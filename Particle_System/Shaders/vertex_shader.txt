#version 330 core
layout (location = 0) in vec3 aPos;
layout (location = 1) in vec3 aColor;

uniform float zoom_value;

uniform mat4 model;
uniform mat4 view;
uniform mat4 projection;

out vec3 vColor;
void main()
{
 gl_Position = projection * view * model * vec4(aPos.x * zoom_value, aPos.y * zoom_value, aPos.z, 1.0);
 vColor = aColor;
}
